! !! RR1

enable
configure terminal

hostname RR1

interface fa 0/0
ip address 16.04.19.61 255.255.255.252
duplex full
speed 100
no shutdown
exit

do show ip interface brief

interface fa 0/1
ip address 16.04.01.61 255.255.255.252
duplex full
speed 100
no shutdown
exit

do show ip interface brief

router rip
version 2
no auto-summary
network 16.04.19.60
network 16.04.01.60
passive-interface default
no passive-interface fa 0/0
no passive-interface fa 0/1
exit

do show ip interface brief
do show ip route
do show ip protocols

exit
