! !! SA-SW31 Configuration (Layer 2 Switch with VLANs)

enable

vlan database
vlan 311
vlan 321
apply
exit

configure terminal

hostname SA-SW31

interface FastEthernet1/0
description Trunk to OR4 Gi0/0
switchport trunk encapsulation dot1q
switchport mode trunk
switchport trunk allowed vlan 311,321
no shutdown
exit

! Configure access port for SA-PC311 (VLAN 311)
interface FastEthernet1/1
description SA-PC311
switchport mode access
switchport access vlan 311
no shutdown
exit

! Configure access port for SA-PC321 (VLAN 321)
interface FastEthernet1/2
description SA-PC321
switchport mode access
switchport access vlan 321
no shutdown
exit

do show vlan-switch brief
do show interfaces trunk

end
write memory
